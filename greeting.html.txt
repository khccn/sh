    <script>
        let selectedTemplate = 1;

        function selectTemplate(num, element) {
            selectedTemplate = num;
            document.querySelectorAll('.template-option').forEach(opt => opt.classList.remove('active'));
            element.classList.add('active');
        }

        function generateAndDownload() {
            const name = document.getElementById('userName').value.trim();
            if(!name) {
                alert('عذراً، الرجاء كتابة الاسم أولاً!');
                return;
            }

            const canvas = document.getElementById('cardCanvas');
            const ctx = canvas.getContext('2d');
            const img = new Image();
            img.crossOrigin = "Anonymous"; 
            
            // اختيار ملف الصورة بناءً على الاختيار
            img.src = selectedTemplate === 1 ? 'template1.jpg' : 'template2.jpg'; 

            img.onload = () => {
                canvas.width = img.width;
                canvas.height = img.height;
                ctx.drawImage(img, 0, 0);

                // إعدادات النص
                ctx.font = "bold 80px 'Tajawal'"; 
                ctx.fillStyle = "#7A5230"; 
                ctx.textAlign = "center";
                ctx.textBaseline = "middle";

                // توزيع النص: يمكنك تغيير هذه القيم لكل تصميم إذا أردت
                let yPosition = canvas.height / 2 + 150; 
                
                // مثال: إذا كان التصميم الثاني يحتاج مكان مختلف للاسم
                if(selectedTemplate === 2) {
                    yPosition = canvas.height / 2 + 200; 
                }

                ctx.fillText(name, canvas.width / 2, yPosition);

                const link = document.createElement('a');
                link.download = `تهنئة-شادن-${name}.jpg`;
                link.href = canvas.toDataURL('image/jpeg', 0.9);
                link.click();
            };

            img.onerror = () => {
                alert('خطأ: تأكد من رفع ملفات template1.jpg و template2.jpg على GitHub');
            };
        }
    </script>
